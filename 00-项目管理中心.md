## ⏳ 项目时间线 (Gantt)

```mermaid
gantt
    title 项目计划全景图
    dateFormat YYYY-MM-DD
    axisFormat %m-%d
    
    %% 这里使用 DataviewJS 动态生成 mermaid 代码
    %% 请确保安装了 Dataview 插件并启用了 JS 权限
```

```dataviewjs
const projects = dv.pages("30-业务档案/00-项目库").where(p => p.status == "进行中" || p.status == "阻塞");

if (projects.length == 0) {
    dv.paragraph("暂无进行中的项目");
} else {
    let mermaidText = "gantt\n    title 进行中项目概览\n    dateFormat YYYY-MM-DD\n    axisFormat %m-%d\n\n";
    
    for (let p of projects) {
        let title = p.file.name;
        let start = p.startDate ? moment(p.startDate.toString()).format("YYYY-MM-DD") : moment().format("YYYY-MM-DD");
        // 如果没有结束时间，默认加30天
        let end = p.endDate ? moment(p.endDate.toString()).format("YYYY-MM-DD") : moment(start).add(30, 'days').format("YYYY-MM-DD");
        
        mermaidText += `    section ${title}\n`;
        mermaidText += `    ${p.status} :${p.status == "完成" ? "done," : "active," } ${start}, ${end}\n`;
    }
    
    dv.paragraph("```mermaid\n" + mermaidText + "\n```");
}
```

## 🏗️ 活跃项目列表

```dataview
TABLE 
    manager as "负责人",
    priority as "优先级",
    endDate as "截止日期",
    length(file.inlinks) as "关联文档数"
FROM "30-业务档案/00-项目库"
WHERE status != "归档" AND status != "已完成"
SORT priority desc
```

## 📈 整体任务分布

```dataviewjs
// 统计所有项目库里的项目
let pages = dv.pages("30-业务档案/00-项目库");
let active = pages.where(p => p.status == "进行中").length;
let blocked = pages.where(p => p.status == "阻塞").length;
let notStarted = pages.where(p => p.status == "未开始").length;

dv.paragraph("
- 🟢 **进行中**: ${active}
- 🔴 **阻塞**: ${blocked}
- ⚪ **未开始**: ${notStarted}
");
```
