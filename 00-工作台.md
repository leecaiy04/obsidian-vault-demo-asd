## 🔥 焦点任务 (Focus)

```dataviewjs
// 获取所有未归档的业务文档
let pages = dv.pages('"30-业务档案"').where(p => p.status != "归档" && p.status != "取消");

// 筛选：高优先级 或 正在进行
let focusPages = pages.where(p => p.priority == "高" || p.status == "进行中");

dv.table(
    ["任务/项目", "状态", "优先级", "截止日期", "领域"],
    focusPages
    .sort(p => p.priority == "高" ? 0 : 1) // 高优先级排前面
    .map(p => [
        p.file.link, 
        p.status, 
        p.priority, 
        p.dueDate ? p.dueDate : "-", 
        p.area
    ])
)
```

## 📅 待办聚合 (Global Tasks)

```dataview
TASK
FROM ("30-业务档案" OR "20-日志") AND -"99-模板"
WHERE !completed 
  AND status != "归档"
  AND text != "" 
  AND text != " "
  AND !contains(text, "今日新增")
GROUP BY file.link
```

## 🕒 最近更新 (Recently Updated)

```dataview
TABLE status, area, updated as "更新时间"
FROM "30-业务档案"
WHERE file.mtime >= date(today) - dur(3 days)
SORT file.mtime desc
LIMIT 10
```

## 🔍 高级内容检索 (Omnisearch + Dataview)

> [!tip]- 修改下方代码中的 `keyword` 进行搜索
> 此视图结合了全文检索和属性过滤，能找到**内容包含关键词**且**状态未归档**的业务文档。

```dataviewjs
// 👇 在这里修改关键词
const keyword = "修改此处搜索";

if (keyword === "修改此处搜索") {
    dv.paragraph("请在代码块中设置 `keyword` 变量来开始搜索。");
} else {
    // 1. 全文检索 (找内容)
    const results = await omnisearch.search(keyword);
    const matchedPaths = results.map(r => r.path);

    // 2. 属性过滤 (找状态)
    const pages = dv.pages('"30-业务档案"')
        .where(p => matchedPaths.includes(p.file.path)) // 必须是搜到的文件
        .where(p => p.status != "归档");               // 且未归档

    if (pages.length == 0) {
        dv.paragraph(`未找到内容包含 "**${keyword}**" 的活跃文档`);
    } else {
        dv.header(4, `包含 "${keyword}" 的活跃文档 (${pages.length})`);
        dv.table(
            ["文档", "状态", "更新时间"],
            pages.sort(p => p.updated, "desc")
            .map(p => [p.file.link, p.status, p.updated])
        );
    }
}
```
